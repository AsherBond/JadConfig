4    * Interface for configuration repositories
5    * <p/>
6    * A configuration repository can be any data source containing configuration data, e. g. a simple in-memory store,
7    * a file, or a database.
8    *
9    * @author jschalanda
10   */
11  public interface Repository {
12  
13      /**
14       * Opens the configuration repository, e. g. create a database connection, open a file on disk for reading
15       *
16       * @throws RepositoryException If an error occurred while opening the data source
17       */
18      void open() throws RepositoryException;
19  
20      /**
21       * Reads the configuration parameter {@literal name} from the backing data source. The {@literal name} is specific
22       * to the underlying data source.
23       *
24       * @param name The parameter name
25       * @return The value of the provided {@literal name} as {@link String}
26       */
27      String read(String name);
28  
29      /**
30       * Writes the value {@literal value} for configuration parameter {@literal name} to the underlying data source.
31       * <p/>
32       * Depending on the data source this method might also just collect changed configuration parameters and finally
33       * write them to the underlying data source when the {@link #save()} method is being called.
34       *
35       * @param name  The parameter name
36       * @param value The parameter value
37       * @throws RepositoryException If an error occurred while writing to the data source
38       * @see #save()
39       */
40      void write(String name, String value) throws RepositoryException;
41  
42      /**
43       * Saves any changed configuration parameters to the underlying data source.
44       *
45       * @throws RepositoryException If an error occurred while writing to the data source
46       * @see #write(String, String)
47       */
48      void save() throws RepositoryException;
49  
50      /**
51       * Closes the underlying data source when it isn't require any more.
52       *
53       * @throws RepositoryException If an error occurred while closing the underlying data source

55      void close() throws RepositoryException;
